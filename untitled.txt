<script>
let balance = localStorage.getItem("balance") || 100;
document.getElementById("balance").innerText = balance;

// 🔐 LOGIN SYSTEM
function login(){
let user = document.getElementById("username").value;
let pass = document.getElementById("password").value;

if(user !== "" && pass !== ""){
document.getElementById("landing").style.display="none";
document.getElementById("app").style.display="block";
}else{
alert("Enter details");
}
}

// 💰 WALLET
function deposit(){
let amt = parseInt(document.getElementById("amount").value);
balance = parseInt(balance) + amt;
update();
}

function withdraw(){
let amt = parseInt(document.getElementById("amount").value);
if(balance >= amt){
balance -= amt;
update();
}else{
alert("Low balance");
}
}

// 🎮 GAMES
function spin(){
let win = Math.floor(Math.random()*50);
balance += win;
document.getElementById("result").innerText="🎡 You won ₹"+win;
update();
}

function cricket(){
let score = Math.floor(Math.random()*100);
if(score > 50){
balance += 20;
document.getElementById("result").innerText="🏏 Win Score "+score;
}else{
balance -= 10;
document.getElementById("result").innerText="🏏 Lose Score "+score;
}
update();
}

function rocket(){
let multi = (Math.random()*5).toFixed(2);
let win = Math.floor(multi*10);
balance += win;
document.getElementById("result").innerText="🚀 x"+multi+" = ₹"+win;
update();
}

// 🔄 UPDATE
function update(){
document.getElementById("balance").innerText = balance;
localStorage.setItem("balance", balance);
}
